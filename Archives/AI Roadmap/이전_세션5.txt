def load_universe(path: Path = UNIVERSE_CSV) -> pd.DataFrame:
    if not path.exists():
        raise FileNotFoundError(f"Universe CSV not found: {path}")
    df = pd.read_csv(path, encoding="utf-8-sig")
    need = {"구분", "순위", "종목명", "Ticker"}
    if not need.issubset(df.columns):

    log.info(f"Universe loaded: {len(df)} tickers")

def fetch_yf(ticker: str, start: str, end: str) -> pd.DataFrame:
    try:
        df = yf.download(ticker, start=start, end=end, progress=False)
        if df.empty:
            log.warning(f"[YF] No data for {ticker}")
            return pd.DataFrame()
        df = df[REQUIRED_COLS].copy()
        df.index.name = "Date"
        return df
    except Exception as e:
        log.error(f"[YF] Error for {ticker}: {e}")
        return pd.DataFrame()

def fetch_krx(ticker: str, start: str, end: str) -> pd.DataFrame:
    if stock is None:
        log.warning("[KRX] pykrx not installed")
        return pd.DataFrame()
    try:
        code = ticker.split(".")[0]
        df = stock.get_market_ohlcv_by_date(start.replace("-", ""), end.replace("-", ""), code)
        if df.empty:
            return pd.DataFrame()
        df.rename(columns={"시가":"Open","고가":"High","저가":"Low","종가":"Close","거래량":"Volume"}, inplace=True)
        df.index = pd.to_datetime(df.index)
        df.index.name = "Date"
        return df[REQUIRED_COLS]
    except Exception as e:
        log.error(f"[KRX] Error for {ticker}: {e}")
        return pd.DataFrame()

# -----------------------
# Validation & Cache
# -----------------------
def validate(df: pd.DataFrame) -> bool:
    if df.empty:
        return False
    if not set(REQUIRED_COLS).issubset(df.columns):
        return False
    if df.isna().any().any():
        return False
    return True

def cache_path(ticker: str, start: str, end: str) -> Path:
    return CACHE_DIR / f"{ticker}_{start}_{end}.csv"

def load_cache(path: Path) -> pd.DataFrame:
    if path.exists():
        try:
            df = pd.read_csv(path, parse_dates=["Date"], index_col="Date")
            log.info(f"[CACHE] Hit: {path.name}")
            return df
        except Exception as e:
            log.warning(f"[CACHE] Read fail {path.name}: {e}")
    return pd.DataFrame()

def save_cache(df: pd.DataFrame, path: Path):
    df.to_csv(path, encoding="utf-8-sig")
    log.info(f"[CACHE] Saved: {path.name} ({len(df):,} rows)")

# -----------------------
# Orchestrator
# -----------------------
def process_ticker(ticker: str, start: str, end: str, source: str, force: bool=False):
    cpath = cache_path(ticker, start, end)
    df = pd.DataFrame()
    if not force:
        df = load_cache(cpath)

    if df.empty:
        if source == "yfinance":
            df = fetch_yf(ticker, start, end)
        elif source == "pykrx":
            df = fetch_krx(ticker, start, end)

        if not validate(df):
            log.warning(f"[{ticker}] invalid → fallback KRX")
            df = fetch_krx(ticker, start, end)

        if validate(df):
            save_cache(df, cpath)
        else:
            log.error(f"[{ticker}] FAIL: no valid data")
    else:
        log.info(f"[{ticker}] using cache")

def run(start: str, end: str, source: str, force: bool):
    uni = load_universe()
    for _, row in uni.iterrows():
        ticker = str(row["Ticker"]).strip()
        process_ticker(ticker, start, end, source, force)

REQUIRED_COLS = ["Open", "High", "Low", "Close", "Volume"]

# -----------------------
# Universe
# -----------------------
def load_universe(path: Path = UNIVERSE_CSV) -> pd.DataFrame:
    if not path.exists():
        raise FileNotFoundError(f"[UNIVERSE] Not found: {path}")
    df = pd.read_csv(path, encoding="utf-8-sig")
    if df.empty:
        raise ValueError("[UNIVERSE] Empty CSV")
    need = {"구분", "순위", "종목명", "Ticker"}
    if not need.issubset(df.columns):
        raise ValueError(f"[UNIVERSE] Schema mismatch. required={need}, got={set(df.columns)}")
    df["Ticker"] = df["Ticker"].astype(str).str.strip()
    log.info(f"[UNIVERSE] Loaded: {len(df)} tickers from {path}")
    return df

# -----------------------
# Fetchers (+소규모 재시도)
# -----------------------
def _retry(times=2):
    def deco(fn):
        def wrap(*a, **kw):
            last = None
            for i in range(times+1):
                try:
                    return fn(*a, **kw)
                except Exception as e:
                    last = e
                    log.warning(f"[RETRY] {fn.__name__} attempt {i+1}/{times+1} failed: {e}")
            raise last
        return wrap
    return deco

@_retry(times=1)
def fetch_yf(ticker: str, start: str, end: str) -> pd.DataFrame:
    df = yf.download(ticker, start=start, end=end, progress=False, auto_adjust=False)
    if df is None or df.empty:
        log.warning(f"[YF] Empty: {ticker}")
        return pd.DataFrame()
    # 표준 스키마 맞춤
    cols = [c for c in REQUIRED_COLS if c in df.columns]
    if len(cols) < len(REQUIRED_COLS):
        log.warning(f"[YF] Missing columns for {ticker}: have={df.columns.tolist()}")
        return pd.DataFrame()
    df = df[REQUIRED_COLS].copy()
    # tz 제거 및 인덱스 정리
    if getattr(df.index, "tz", None) is not None:
        df.index = df.index.tz_localize(None)
    df.index.name = "Date"
    return df

@_retry(times=1)
def fetch_krx(ticker: str, start: str, end: str) -> pd.DataFrame:
    if stock is None:
        log.warning("[KRX] pykrx not installed; fallback unavailable")
        return pd.DataFrame()
    code = ticker.split(".")[0]
    s = start.replace("-", "")
    e = end.replace("-", "")
    df = stock.get_market_ohlcv_by_date(s, e, code)
    if df is None or df.empty:
        log.warning(f"[KRX] Empty: {ticker}")
        return pd.DataFrame()
    df = df.rename(columns={"시가":"Open","고가":"High","저가":"Low","종가":"Close","거래량":"Volume"})
    # 컬럼 누락 방지
    miss = [c for c in REQUIRED_COLS if c not in df.columns]
    if miss:
        log.warning(f"[KRX] Missing columns for {ticker}: {miss}")
        return pd.DataFrame()
    df.index = pd.to_datetime(df.index)
    df.index.name = "Date"
    df = df[REQUIRED_COLS].copy()
    return df

# -----------------------
# Validation & Cache
# -----------------------
def validate(df: pd.DataFrame) -> bool:
    if df is None or df.empty:
        return False
    if not set(REQUIRED_COLS).issubset(df.columns):
        return False
    if df.isna().any().any():
        return False
    return True

def cache_path(ticker: str, start: str, end: str) -> Path:
    safe = ticker.replace("/", "_").replace("\\", "_").strip()
    return CACHE_DIR / f"{safe}_{start}_{end}.csv"

def load_cache(path: Path) -> pd.DataFrame:
    if path.exists():
        try:
            df = pd.read_csv(path, parse_dates=["Date"], index_col="Date")
            if getattr(df.index, "tz", None) is not None:
                df.index = df.index.tz_localize(None)
            log.info(f"[CACHE] Hit: {path.name}")
            return df
        except Exception as e:
            log.warning(f"[CACHE] Read fail ({path.name}): {e}")
    return pd.DataFrame()

def save_cache(df: pd.DataFrame, path: Path):
    df.to_csv(path, encoding="utf-8-sig")
    log.info(f"[CACHE] Saved: {path.name} ({len(df):,} rows)")

# -----------------------
# Orchestrator
# -----------------------
def process_ticker(ticker: str, start: str, end: str, source: str, force: bool=False):
    cpath = cache_path(ticker, start, end)
    df = pd.DataFrame()
    if not force:
        df = load_cache(cpath)

    if df.empty:
        if source == "yfinance":
            df = fetch_yf(ticker, start, end)
        elif source == "pykrx":
            df = fetch_krx(ticker, start, end)

        if not validate(df):
            log.warning(f"[{ticker}] invalid from {source} → fallback KRX")
            df = fetch_krx(ticker, start, end)

        if validate(df):
            save_cache(df, cpath)
        else:
            log.error(f"[{ticker}] FAIL: no valid data")
    else:
        log.info(f"[{ticker}] using cache")

def run(start: str, end: str, source: str, force: bool):
    uni = load_universe()
    for _, row in uni.iterrows():
        ticker = str(row["Ticker"]).strip()
        if not ticker:
            log.warning("[UNIVERSE] empty ticker row skipped")
            continue
        process_ticker(ticker, start, end, source, force)

def parse_args():
    today = datetime.today().strftime("%Y-%m-%d")
    p = argparse.ArgumentParser(description="Endeavour Data Handler")
    p.add_argument("--source", choices=["yfinance","pykrx"], default="yfinance")
    p.add_argument("--start", default="2020-01-01")
    p.add_argument("--end", default=today)
    p.add_argument("--force", action="store_true")
    return p.parse_args()

if __name__ == "__main__":
    args = parse_args()
    log.info(f"RUN source={args.source} range={args.start}..{args.end} force={args.force}")
    try:
        run(args.start, args.end, args.source, args.force)
    except Exception as e:
        log.exception(f"FATAL: {e}")
